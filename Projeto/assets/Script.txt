CREATE TABLE empresa
(
	empresaid Serial PRIMARY KEY ,
	razaosocial char(100),
	cnpj char(20),
	cep char(10),
	site char(100),
	telefone char(20),
	endereco char(100),
	cidade char(30),
	estado char(25),
	situacao smallint,
	foto character varying(100),
	datacadastro date,
	usuario integer default 0
);


CREATE TABLE usuario
(
	usuarioid Serial PRIMARY KEY,
	email char(100),
	senha text
);

CREATE TABLE promocoes
(
	promocaoid Serial PRIMARY KEY,
	descricaopromocao char(100),
	datavalidade date,
	situacao smallint,
	foto bit varying,
	empresa integer default 0
);

CREATE TABLE cliente
(
	clienteid Serial PRIMARY KEY,
	nome char(90),
	cep char(10),
	cpf char(20)
);

CREATE TABLE empresa_cliente
(
	empresaid integer PRIMARY KEY,
	cnpj char(20),
	clienteid integer,
	cpf char(20)
);


CREATE TABLE empresa_promocao
(
	empresaid integer PRIMARY KEY,
	promocaoid integer,
	razaosocial char(100),
	descricaopromocao char(100)
);

CREATE TABLE IF NOT EXISTS  ci_sessions (
    sequencial_id Serial NOT NULL,
    sessao_id varchar(40) ,
    endereco_ip varchar(16) DEFAULT '0',
    user_agent varchar(50) ,
    ultima_atividade integer DEFAULT 0,
    usuario_id integer,
    email varchar(100),
    senha text,
   	empresa_id integer,
    empresa_nome varchar(100),
    PRIMARY KEY (sequencial_id)
);
